\begin{frame}{Konwersja z systemu dziesiÍtnego}
	Jak przekszta≥ciÊ liczbÍ naturalnπ $n$ zapisanπ w systemie o podstawie $10$ na liczbÍ~$n$ zapisanπ w systemie o podstawie $b$?
	\begin{block}{Konwersja z systemu dziesiÍtnego: Pierwszy sposÛb}

		\end{enumerate}
	\end{block}
\end{frame}

	\begin{exampleblock}{Przyk≥ad}
